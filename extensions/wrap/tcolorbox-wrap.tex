\usepackage{tcolorbox}
\tcbuselibrary{most}
\usepackage{fontawesome5}
\newtcolorbox{info}[1][\infoTitle]{%
  enlarge top by=0.3cm,
  enhanced jigsaw, % better frame drawing
  borderline west={2pt}{0pt}{infoBorder}, % straight vertical line at the left edge
  %sharp corners, % No rounded corners
  boxrule=0pt, % no real frame,
  fonttitle={\large\bfseries},
  coltitle={black},  % Black colour for title
  colbacktitle=info,
  arc=4pt,
  breakable,
  title={\infoTitleFa#1~~},  % Fixed title
  attach title, % Move the title into the box
}
